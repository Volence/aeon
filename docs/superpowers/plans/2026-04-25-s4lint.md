**Goal:** Build a Python static analysis tool (`tools/s4lint.py`) that catches 68000 assembly correctness bugs at build time, integrated into the aeon build pipeline.

**Architecture:** Single Python script with three layers: tokenizer (line → structured tuple), check functions (each implements one E/W code), and a state tracker (Z80 bus, alignment, SR save per routine). Tests use Python's built-in `unittest`. The linter is invoked by `build.sh` before the assembler.

**Tech Stack:** Python 3.14 (stdlib only — no pip dependencies), `unittest` for tests.

---

## File Structure

| File | Responsibility |
|---|---|
| `tools/s4lint.py` | Main linter script — CLI, file discovery, tokenizer, state tracker, all checks, diagnostic output |
| `tools/test_s4lint.py` | Unit tests — one test class per check code, using inline assembly strings |
| `build.sh` | Modified to call s4lint before assembler |

The entire linter lives in one file. It's ~800 lines of Python, not a framework. Tests in a sibling file. No config file — the macro contract registry and SST constants are dicts at the top of `s4lint.py`.

```python
#!/usr/bin/env python3
"""s4lint — static analysis for 68000 assembly (Sonic 4 Engine)."""
import sys
import os
import re
import argparse
from collections import namedtuple
from typing import Optional

Token = namedtuple("Token", ["label", "instruction", "size", "operands", "comment"])

# Assembler directives and pseudo-ops (recognized but not 68000 instructions)
ASM_DIRECTIVES = frozenset({
    "include", "binclude", "incbin", "if", "ifdef", "ifndef", "else", "elseif",
    "endif", "rept", "endr", "phase", "dephase", "org", "align", "even", "odd",
    "cpu", "padding", "supmode", "end", "error", "warning", "set", "eval",
    "struct", "endstruct", "ds", "dc", "function", "macro", "endm",
    "irp", "irpc", "section", "endsection",
})

# 68000 branch/jump mnemonics that require sizing
BRANCH_MNEMONICS = frozenset({
    "bra", "bsr", "beq", "bne", "bhi", "blo", "bhs", "bls",
    "bge", "bgt", "ble", "blt", "bpl", "bmi", "bcc", "bcs",
    "bvc", "bvs",
})

# Regex for parsing assignment lines: LABEL = VALUE or LABEL equ VALUE
RE_ASSIGNMENT = re.compile(
    r'^(\w+)\s*=\s*(.+?)(?:\s*;.*)?$'
)

# Regex for a label at the start of a line
RE_LABEL = re.compile(
    r'^(\.\w+|[A-Za-z_]\w*):?\s*(.*)'
)

# Regex for instruction with optional size suffix
RE_INSTRUCTION = re.compile(
    r'^(\w+)(\.[bBwWlLsS])?\s*(.*)'
)


def tokenize_line(line: str) -> Token:
    """Parse one line of 68000 assembly into a Token."""
    label = None
    instruction = None
    size = None
    operands = []
    comment = None

    # Strip trailing newline/whitespace
    raw = line.rstrip()

    # Extract comment
    comment_pos = _find_comment(raw)
    if comment_pos is not None:
        comment = raw[comment_pos + 1:]
        raw = raw[:comment_pos].rstrip()

    # Empty after stripping
    if not raw.strip():
        return Token(label, instruction, size, operands, comment)

    rest = raw

    # Check for assignment: LABEL = EXPR
    m = RE_ASSIGNMENT.match(rest)
    if m:
        return Token(m.group(1), "=", None, [m.group(2).strip()], comment)

    # Check for label at start of line (no leading whitespace or local .label)
    stripped = rest.lstrip()
    if rest and not rest[0].isspace():
        m = RE_LABEL.match(rest)
        if m:
            label = m.group(1).rstrip(":")
            rest = m.group(2).strip()
            if not rest:
                return Token(label, instruction, size, operands, comment)
    elif stripped.startswith(".") and ":" in stripped.split()[0] if stripped else False:
        # Local label like .loop: at indented position
        m = RE_LABEL.match(stripped)
        if m:
            label = m.group(1).rstrip(":")
            rest = m.group(2).strip()
            if not rest:
                return Token(label, instruction, size, operands, comment)
    else:
        rest = stripped

    if not rest:
        return Token(label, instruction, size, operands, comment)

    # Parse instruction
    m = RE_INSTRUCTION.match(rest)
    if m:
        instruction = m.group(1)
        if m.group(2):
            size = m.group(2).lower()
        operand_str = m.group(3).strip()

        # Preserve original case for certain directives
        instr_lower = instruction.lower()
        if instr_lower in ASM_DIRECTIVES:
            if instr_lower not in ("binclude",):
                instruction = instr_lower
        # Don't lowercase macro names or unknown identifiers — they need
        # case-sensitive matching against the macro contract registry.
        # But lowercase known 68000 mnemonics.
        elif _is_68k_mnemonic(instr_lower):
            instruction = instr_lower

        # Handle "label struct" / "label endstruct" patterns
        if instruction and instruction.lower() in ("struct", "endstruct"):
            # The label was actually parsed as the instruction, and struct is the operand
            # Re-parse: this is "SST struct" where SST is the label
            pass  # Already handled — label was captured above

        if operand_str:
            operands = _split_operands(operand_str)

    return Token(label, instruction, size, operands, comment)


def _find_comment(line: str) -> Optional[int]:
    """Find the position of ';' that starts a comment (not inside a string)."""
    in_string = False
    for i, ch in enumerate(line):
        if ch == '"':
            in_string = not in_string
        elif ch == ";" and not in_string:
            return i
    return None


def _split_operands(s: str) -> list:
    """Split operand string by commas, respecting parentheses."""
    parts = []
    depth = 0
    current = []
    for ch in s:
        if ch == "(" :
            depth += 1
            current.append(ch)
        elif ch == ")":
            depth -= 1
            current.append(ch)
        elif ch == "," and depth == 0:
            parts.append("".join(current).strip())
            current = []
        else:
            current.append(ch)
    if current:
        parts.append("".join(current).strip())
    return parts

def _is_68k_mnemonic(s: str) -> bool:
    return s in _M68K_MNEMONICS

# Diagnostic infrastructure

class Diagnostic:
    __slots__ = ("file", "line", "severity", "code", "message")

    def __init__(self, file, line, severity, code, message):
        self.file = file
        self.line = line
        self.severity = severity
        self.code = code
        self.message = message

    def __str__(self):
        return f"{self.file}:{self.line}: {self.severity}: {self.message} [{self.code}]"


class LintContext:
    """Per-file linting state."""

    def __init__(self, filepath, options=None):
        self.filepath = filepath
        self.options = options or {}
        self.diagnostics = []

        # State tracking (reset at routine boundaries)
        self.z80_state = "unknown"          # "running", "stopped", "unknown"
        self.z80_stopped_line = None
        self.ints_state = "unknown"         # "enabled", "disabled", "unknown"
        self.sr_saved = False
        self.sr_saved_line = None
        self.byte_count = 0                 # bytes since last even/align
        self.in_dbf_loop = False
        self.dbf_loop_label = None
        self.current_routine = None         # current global label
        self.prev_instruction = None        # previous Token (for 2-line patterns)
        self.local_labels = set()           # local labels seen in current routine
        self.routine_lines = []             # lines in current routine (for header check)
        self.last_routine_terminated = True # did prev routine end with rts/rte/bra?
        self.in_struct = False
        self.in_rept = 0                    # nesting depth
        self.in_macro_def = False

    def emit(self, line, severity, code, message):
        self.diagnostics.append(Diagnostic(self.filepath, line, severity, code, message))

    def error(self, line, code, message):
        self.emit(line, "error", code, message)

    def warning(self, line, code, message):
        self.emit(line, "warning", code, message)

    def reset_routine_state(self):
        """Reset state at a new routine boundary."""
        self.z80_state = "unknown"
        self.z80_stopped_line = None
        self.ints_state = "unknown"
        self.sr_saved = False
        self.sr_saved_line = None
        self.byte_count = 0
        self.in_dbf_loop = False
        self.dbf_loop_label = None
        self.local_labels = set()
        self.routine_lines = []

    def check_routine_end(self, line_num):
        """Check for state violations at routine end (rts/rte)."""
        if self.z80_state == "stopped":
            self.error(line_num, "E007",
                f"stopZ80 at line {self.z80_stopped_line} has no matching startZ80 before rts")
        if self.sr_saved:
            self.error(line_num, "E010",
                f"sr saved at line {self.sr_saved_line} but not restored before rts")

def discover_files(entry_path: str, follow_includes: bool = True) -> list:
    """Build ordered file list by following include directives."""
    if not follow_includes:
        return [entry_path]

    base_dir = os.path.dirname(os.path.abspath(entry_path))
    files = []
    seen = set()

    def _scan(filepath):
        abspath = os.path.abspath(filepath)
        if abspath in seen:
            return
        seen.add(abspath)
        files.append(abspath)

        try:
            with open(abspath, "r") as f:
                for line in f:
                    t = tokenize_line(line)
                    if t.instruction in ("include",) and t.operands:
                        inc_path = t.operands[0].strip('"')
                        inc_full = os.path.join(base_dir, inc_path)
                        if os.path.isfile(inc_full):
                            _scan(inc_full)
        except FileNotFoundError:
            pass

    _scan(entry_path)
    return files

# CLI

SKIP_PATHS = {"debug/debugger.asm", "debug/error_handler.asm"}

def parse_args(argv=None):
    p = argparse.ArgumentParser(description="s4lint — 68000 assembly linter")
    p.add_argument("files", nargs="+", help="Assembly files to lint (or main.asm to follow includes)")
    p.add_argument("--warnings-as-errors", action="store_true", help="Treat warnings as errors")
    p.add_argument("--no-warnings", action="store_true", help="Suppress all warnings")
    p.add_argument("--only", type=str, default=None, help="Only run these checks (comma-separated)")
    p.add_argument("--skip", type=str, default=None, help="Skip these checks (comma-separated)")
    p.add_argument("--no-follow-includes", action="store_true", help="Don't follow include directives")
    return p.parse_args(argv)


def main(argv=None):
    args = parse_args(argv)

    options = {
        "warnings_as_errors": args.warnings_as_errors,
        "no_warnings": args.no_warnings,
        "only": set(args.only.split(",")) if args.only else None,
        "skip": set(args.skip.split(",")) if args.skip else None,
    }

    # Discover files
    all_files = []
    for f in args.files:
        if not args.no_follow_includes and os.path.basename(f) == "main.asm":
            all_files.extend(discover_files(f, follow_includes=True))
        else:
            all_files.append(os.path.abspath(f))

    # Filter out skip paths
    base_dir = os.path.dirname(os.path.abspath(args.files[0]))
    filtered = []
    for fp in all_files:
        rel = os.path.relpath(fp, base_dir)
        if rel.replace(os.sep, "/") not in SKIP_PATHS:
            filtered.append(fp)

    has_errors = False
    has_warnings = False
    for fp in filtered:
        ctx = lint_file(fp, options, base_dir)
        for d in ctx.diagnostics:
            # Check suppression
            if _is_suppressed(d.code, options):
                continue
            if d.severity == "warning" and options.get("no_warnings"):
                continue
            print(d, file=sys.stderr)
            if d.severity == "error":
                has_errors = True
            else:
                has_warnings = True

    if has_errors or (has_warnings and options.get("warnings_as_errors")):
        sys.exit(1)
    sys.exit(0)


def _is_suppressed(code, options):
    if options.get("only") and code not in options["only"]:
        return True
    if options.get("skip") and code in options["skip"]:
        return True
    return False


def lint_file(filepath, options=None, base_dir=None):
    """Lint a single file. Returns LintContext with diagnostics."""
    ctx = LintContext(
        os.path.relpath(filepath, base_dir) if base_dir else filepath,
        options
    )

    try:
        with open(filepath, "r") as f:
            lines = f.readlines()
    except FileNotFoundError:
        ctx.error(0, "E000", f"File not found: {filepath}")
        return ctx

    for line_num, raw_line in enumerate(lines, 1):
        token = tokenize_line(raw_line)

        # Check for inline suppression
        suppressed = set()
        if token.comment and "lint: disable=" in token.comment:
            m = re.search(r'lint: disable=([A-Z]\d+(?:,[A-Z]\d+)*)', token.comment)
            if m:
                suppressed = set(m.group(1).split(","))

        run_checks(ctx, token, line_num, raw_line, suppressed)
        ctx.prev_instruction = token

    return ctx


def run_checks(ctx, token, line_num, raw_line, suppressed):
    """Run all applicable checks on a single token. Placeholder — tasks 2-7 fill this in."""
    # Track assembler block state
    if token.instruction == "struct":
        ctx.in_struct = True
        return
    if token.instruction == "endstruct":
        ctx.in_struct = False
        return
    if token.instruction == "macro":
        ctx.in_macro_def = True
        return
    if token.instruction == "endm":
        ctx.in_macro_def = False
        return
    if token.instruction == "rept":
        ctx.in_rept += 1
    if token.instruction == "endr":
        ctx.in_rept = max(0, ctx.in_rept - 1)

    # Skip checking inside struct definitions and macro definitions
    if ctx.in_struct or ctx.in_macro_def:
        return

    # Track routine boundaries
    if token.label and not token.label.startswith("."):
        if not ctx.last_routine_terminated:
            pass  # fall-through: carry state forward
        else:
            ctx.reset_routine_state()
        ctx.current_routine = token.label
        ctx.last_routine_terminated = False

    # Track local labels
    if token.label and token.label.startswith("."):
        ctx.local_labels.add(token.label)

    # Track routine termination
    if token.instruction in ("rts", "rte"):
        ctx.check_routine_end(line_num)
        ctx.last_routine_terminated = True


if __name__ == "__main__":
    main()
```

```python
    def _lint(self, line):

        ctx = LintContext("test.asm")
        t = tokenize_line(line)

    def test_unsized_bra(self):
```

```python
RE_DEC = re.compile(r'^(\d+)$')

def parse_numeric(s: str) -> Optional[int]:
    """Try to parse a numeric literal. Returns None if symbolic."""
    s = s.strip().lstrip("#").strip("()")
    # Remove .w/.l suffix
    if s.endswith((".w", ".l", ".W", ".L")):
        s = s[:-2]
    s = s.strip("()")
    m = RE_HEX.match(s)
    if m:

    m = RE_DEC.match(s)
    if m:
        return int(m.group(1))
    return None

        return parse_numeric(inner)
```

```python
# -----------------------------------------------
# E005: Missing alignment after byte data
# -----------------------------------------------

def _count_dc_b_items(operands):
    """Count the number of byte values in a dc.b operand list."""
    count = 0
    for op in operands:
        # Each comma-separated value is one byte, but strings are N bytes
        stripped = op.strip()
        if stripped.startswith('"') and stripped.endswith('"'):
            count += len(stripped) - 2  # subtract quotes
        else:
            count += 1
    return count


def _count_ds_b_items(operands):
    """Parse the count from ds.b N."""
    if operands:
        val = parse_numeric(operands[0])
        if val is not None:
            return val
    return 0


def check_e005_track(ctx, token, line_num):
    """Track byte counts and check alignment. Called from run_checks."""
    # Accumulate byte data
    if token.instruction == "dc" and token.size == ".b":
        ctx.byte_count += _count_dc_b_items(token.operands)
        return
    if token.instruction == "ds" and token.size == ".b":
        ctx.byte_count += _count_ds_b_items(token.operands)
        return

    # Reset on even/align
    if token.instruction in ("even", "align"):
        ctx.byte_count = 0
        return

    # Check: if we have an odd byte count and now see word/long data or an instruction
    if ctx.byte_count > 0 and ctx.byte_count % 2 != 0:
        is_word_data = (
            token.instruction in ("dc", "ds") and token.size in (".w", ".l")
        )
        is_instruction = (
            token.instruction is not None
            and token.instruction not in ASM_DIRECTIVES
            and _is_68k_mnemonic(token.instruction)
        )
        if is_word_data or is_instruction:
            ctx.error(line_num, "E005",
                f"word data/instruction after odd byte count ({ctx.byte_count}) without 'even'")
            ctx.byte_count = 0  # don't re-fire on every subsequent line

    # Non-byte dc/ds resets byte tracking
    if token.instruction in ("dc", "ds") and token.size != ".b":
        ctx.byte_count = 0
```

```python
def check_e006(ctx, token, line_num, suppressed):

        return
```

```python
def _resolve_sst_offset(expr: str) -> Optional[int]:
    """Resolve an SST field expression like SST_sst_custom+28 to a byte offset."""
    m = RE_SST_FIELD_OFFSET.search(expr)
    if not m:
        return None
    field_name = m.group(1)
    base = SST_FIELDS.get(field_name)
    if base is None:
        return None
    extra = int(m.group(2)) if m.group(2) else 0
    return base + extra
```

```python
RE_INDEXED = re.compile(r'\(a[0-7],\s*d[0-7]\.[wl]\)')


def _is_dreg(s):

def _is_areg(s):
    return bool(RE_AREG.match(s.strip().lower()))


def _is_memory_operand(s):
    """Check if operand is a memory reference (not a register)."""
    s = s.strip().lower()
    if RE_DREG.match(s) or RE_AREG.match(s) or s == "sp" or s == "sr":

    return True

    """Parse an immediate operand like #0, #$10, #-1. Returns int or None."""
    s = s.strip()
    if not s.startswith("#"):
        return None
    s = s[1:].strip()
    if s.startswith("-"):
        val = parse_numeric(s[1:])
        return -val if val is not None else None
    return parse_numeric(s)

                if ctx.in_dbf_loop:
                    ctx.warning(line_num, "W010",
                        f"indexed addressing {op} inside loop — pre-compute with lea")

    # W011: movem for single register
    if "W011" not in suppressed and instr == "movem" and ops:
        reg_operand = ops[0] if "(" not in ops[0] else (ops[1] if len(ops) > 1 else "")
        if reg_operand and "-" not in reg_operand and "/" not in reg_operand:
            # Single register
            if RE_DREG.match(reg_operand.strip().lower()) or RE_AREG.match(reg_operand.strip().lower()):
                ctx.warning(line_num, "W011",
                    f"movem with single register {reg_operand} — use move or swap")

    # W012: move.l an,an instead of movea.l
    if "W012" not in suppressed and instr == "move" and size == ".l" and len(ops) == 2:
        if _is_areg(ops[1]):
            ctx.warning(line_num, "W012",
                f"move.l to address register — use movea.l for clarity")

    # W013: move.w #N in moveq range to data register
    if "W013" not in suppressed and instr == "move" and size in (".w", ".l") and len(ops) == 2:
        if _is_dreg(ops[1]):
            imm = _parse_immediate(ops[0])
            if imm is not None and -128 <= imm <= 127 and imm != 0:
                # imm == 0 is already caught by W003
                ctx.warning(line_num, "W013",
                    f"move{size} #{imm}, {ops[1]} — moveq #{imm}, {ops[1]} is faster")
```
