- [ ] **Step 2: Write the block generator skeleton**

Create `tools/ojz_block_gen.py`:

```python
#!/usr/bin/env python3
"""OJZ block-format data generator.

Converts existing column-strip data (generated by ojz_strip_gen.py) into
16x16 tile blocks with embedded collision, independently S4LZ-compressed.

Usage:
    python3 tools/ojz_block_gen.py generate

Input: data/generated/ojz/act1/sec{N}_strips.s4lz (existing column-strip files)
Output: data/generated/ojz/act1/sec{N}_blocks.bin (block index + compressed blocks)
"""

import struct
import sys
import os

sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))
import s4lz

# Constants — must match constants.asm
SECTION_TILE_WIDTH = 256
SECTION_TILE_HEIGHT = 256  # full section height (was 48 for 1D)
BLOCK_SIZE = 16            # 16x16 tiles per block
BLOCKS_PER_AXIS = SECTION_TILE_WIDTH // BLOCK_SIZE  # 16
BLOCKS_PER_SECTION = BLOCKS_PER_AXIS * BLOCKS_PER_AXIS  # 256
STRIP_TILE_HEIGHT = 48     # current strip height (input data)
STRIP_BYTE_SIZE = 128      # bytes per strip column
STRIP_NAMETABLE_SIZE = 96  # 48 words
STRIP_COLLISION_OFFSET = 96
STRIP_COLLISION_ROWS = 24

OUTPUT_DIR = "data/generated/ojz/act1"
NUM_SECTIONS = 9


def load_strip_data(section_idx):
    """Load and decompress a section's column-strip data."""
    path = os.path.join(OUTPUT_DIR, f"sec{section_idx}_strips.s4lz")
    with open(path, "rb") as f:
        compressed = f.read()
    return s4lz.decompress(compressed)


def strips_to_tile_grid(raw_strips):
    """Convert column-strip data to a 2D tile grid + collision grid.

    Returns:
        nametable: list of lists [col][row] of 16-bit tile words
        collision: list of lists [col][collision_row] of 8-bit collision bytes
    """
    num_cols = len(raw_strips) // STRIP_BYTE_SIZE
    nametable = []
    collision = []

    for col_idx in range(num_cols):
        offset = col_idx * STRIP_BYTE_SIZE
        col_tiles = []
        for row in range(STRIP_TILE_HEIGHT):
            word_offset = offset + row * 2
            tile_word = struct.unpack_from(">H", raw_strips, word_offset)[0]
            col_tiles.append(tile_word)
        nametable.append(col_tiles)

        col_coll = []
        for coll_row in range(STRIP_COLLISION_ROWS):
            coll_byte = raw_strips[offset + STRIP_COLLISION_OFFSET + coll_row]
            col_coll.append(coll_byte)
        collision.append(col_coll)

    return nametable, collision


def extract_block(nametable, collision, block_x, block_y):
    """Extract one 16x16 block's nametable + collision data.

    Returns: (nametable_bytes, collision_bytes)
        nametable_bytes: 512 bytes (16x16 tile words, row-major)
        collision_bytes: 64 bytes (8x8 collision cells, row-major)
    """
    nt_data = bytearray()
    for row in range(BLOCK_SIZE):
        tile_row = block_y * BLOCK_SIZE + row
        for col in range(BLOCK_SIZE):
            tile_col = block_x * BLOCK_SIZE + col
            if tile_col < len(nametable) and tile_row < len(nametable[tile_col]):
                word = nametable[tile_col][tile_row]
            else:
                word = 0  # air/empty for out-of-range
            nt_data.extend(struct.pack(">H", word))

    coll_data = bytearray()
    coll_rows_per_block = BLOCK_SIZE // 2  # 8 collision rows per 16-tile block
    for coll_row in range(coll_rows_per_block):
        abs_coll_row = block_y * coll_rows_per_block + coll_row
        for col in range(BLOCK_SIZE):
            tile_col = block_x * BLOCK_SIZE + col
            if tile_col < len(collision) and abs_coll_row < len(collision[tile_col]):
                byte = collision[tile_col][abs_coll_row]
            else:
                byte = 0
            coll_data.append(byte)

    return bytes(nt_data), bytes(coll_data)


def generate_section_blocks(section_idx):
    """Generate block index + compressed blocks for one section.

    Output format:
        Block index: 256 x 4-byte ROM pointers (relative to data start)
        Followed by: concatenated S4LZ-compressed blocks
    """
    raw_strips = load_strip_data(section_idx)
    nametable, collision = strips_to_tile_grid(raw_strips)

    # Extract and compress all 256 blocks
    compressed_blocks = []
    for block_y in range(BLOCKS_PER_AXIS):
        for block_x in range(BLOCKS_PER_AXIS):
            nt_data, coll_data = extract_block(nametable, collision, block_x, block_y)
            raw_block = nt_data + coll_data  # 576 bytes
            if all(b == 0 for b in raw_block):
                compressed_blocks.append(None)  # null block (all air)
            else:
                compressed_blocks.append(s4lz.compress(raw_block))

    # Build output: index table + data
    index_size = BLOCKS_PER_SECTION * 4  # 1024 bytes
    data_offset = index_size
    index_table = bytearray()
    block_data = bytearray()

    for block in compressed_blocks:
        if block is None:
            index_table.extend(struct.pack(">I", 0))  # null pointer
        else:
            # Offset relative to start of this section's block data blob
            index_table.extend(struct.pack(">I", data_offset + len(block_data)))
            block_data.extend(block)
            # Word-align
            if len(block_data) % 2:
                block_data.append(0)

    return bytes(index_table) + bytes(block_data)


def generate_all():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    for sec_idx in range(NUM_SECTIONS):
        print(f"Section {sec_idx}...", end=" ", flush=True)
        output = generate_section_blocks(sec_idx)
        out_path = os.path.join(OUTPUT_DIR, f"sec{sec_idx}_blocks.bin")
        with open(out_path, "wb") as f:
            f.write(output)
        print(f"{len(output)} bytes")

    print("Done.")


if __name__ == "__main__":
    if len(sys.argv) < 2 or sys.argv[1] != "generate":
        print("Usage: python3 tools/ojz_block_gen.py generate")
        sys.exit(1)
    generate_all()
```

- [ ] **Step 3: Add block decompression and staging**

Append to `tile_cache.asm`:

```asm
; -----------------------------------------------
; TileCache_DecompressBlock — decompress one 16×16 block into staging buffer
; In:  d0.w = sec_x, d1.w = sec_y, d2.w = block_index (0–255)
; Out: Block_Stage_Nametable and Block_Stage_Collision filled
; Clobbers: d0-d4, a0-a3
; -----------------------------------------------
TileCache_DecompressBlock:
        ; update staging metadata
        move.w  d2, (Block_Stage_ID).w
        move.b  d0, (Block_Stage_Section_X).w
        move.b  d1, (Block_Stage_Section_Y).w

        ; get section descriptor
        movea.l (Current_Act_Ptr).w, a0
        movea.l Act_sec_grid_ptr(a0), a1

        ; sec_id = sec_y * grid_w + sec_x
        move.w  d1, d3
        move.w  Act_grid_w(a0), d4
        ; d3 * d4: grid_w is small (≤16), use add loop or shift if power-of-2
        ; For now use mulu since this runs once per ~16 frames (not per-frame hot path)
        mulu.w  d4, d3
        add.w   d0, d3                         ; d3 = flat section id
        ; d3 × Sec_len (72 = 64+8)
        move.w  d3, d4
        lsl.w   #6, d3
        lsl.w   #3, d4
        add.w   d4, d3
        adda.w  d3, a1                         ; a1 = Sec struct pointer

        ; get block index table
        movea.l Sec_sec_block_index(a1), a0    ; a0 = block index table (ROM)
        cmpa.w  #0, a0
        beq.s   .empty_block

        ; index into block table: block_index × 4
        move.w  d2, d3
        lsl.w   #2, d3
        move.l  (a0, d3.w), d0                 ; d0 = block data pointer (absolute ROM)
        beq.s   .empty_block                   ; null = empty block

        ; decompress block into staging
        movea.l d0, a0
        lea     (Block_Stage_Nametable).l, a2
        moveq   #0, d0
        move.w  #BLOCK_RAW_SIZE, d1
        bsr.w   S4LZ_Decompress                ; one-shot decompress (not streaming)
        rts

.empty_block:
        ; zero-fill staging
        lea     (Block_Stage_Nametable).l, a0
        move.w  #(BLOCK_RAW_SIZE/4)-1, d0
.zero_loop:
        clr.l   (a0)+
        dbf     d0, .zero_loop
        rts
```

```asm
        add.w   d5, d4                         ; d4 = cache_row * 80
        add.w   d1, d4                         ; d4 = cache_row * 80 + cache_col
        add.w   d4, d4                         ; d4 = byte offset
        adda.w  d4, a1

        ; copy nametable words (stride: block=32 bytes/row, cache=160 bytes/row)
        move.w  d3, d4
        subq.w  #1, d4
.copy_nt:
        move.w  (a0), (a1)
        lea     BLOCK_TILE_SIZE*2(a0), a0      ; next block row (32 bytes)
        lea     TILE_CACHE_STRIDE*2(a1), a1    ; next cache row (160 bytes)
        dbf     d4, .copy_nt

        ; collision: same pattern but byte-sized, different arrays
        lea     (Block_Stage_Collision).l, a0
        lsr.w   #1, d0                         ; intra-block col for collision (same as tile col)
        ; collision block stride = 16 bytes/row, cache stride = 80 bytes/row
        ; but collision rows are half tile rows
        ; d0 = intra-block col (0-15), still valid
        adda.w  d0, a0

        lea     (Tile_Cache_Collision).l, a1
        move.w  d2, d4
        lsr.w   #1, d4                         ; cache tile row → collision row
        move.w  d4, d5
        lsl.w   #6, d4
        lsl.w   #4, d5
        add.w   d5, d4                         ; collision_row * 80
        add.w   d1, d4
        adda.w  d4, a1

        move.w  d3, d4
        lsr.w   #1, d4                         ; tile rows → collision rows
        subq.w  #1, d4
        bmi.s   .done_coll
.copy_coll:
        move.b  (a0), (a1)
        lea     BLOCK_TILE_SIZE(a0), a0        ; next collision row in block
        lea     TILE_CACHE_STRIDE(a1), a1      ; next collision row in cache
        dbf     d4, .copy_coll
.done_coll:
        rts

; -----------------------------------------------
; TileCache_CopyBlockRow — copy one row from staging to cache
; In:  d0.w = intra-block row (0–15)
;      d1.w = cache col offset start (world_col_of_block_left - Cache_Left_Col)
;      d2.w = cache row offset (world_row - Cache_Top_Row)
;      d3.w = cols to copy (clamp to cache bounds before calling)
; Out: none
; Clobbers: d0-d5, a0-a2
; -----------------------------------------------
TileCache_CopyBlockRow:
        ; source: Block_Stage_Nametable + (row * 16) * 2
        lea     (Block_Stage_Nametable).l, a0
        move.w  d0, d4
        lsl.w   #5, d4                         ; row * 32 (row * 16 tiles * 2 bytes)
        adda.w  d4, a0

        ; dest: Tile_Cache_Nametable + (cache_row * 80 + cache_col) * 2
        lea     (Tile_Cache_Nametable).l, a1
        move.w  d2, d4
        move.w  d4, d5
        lsl.w   #6, d4
        lsl.w   #4, d5
        add.w   d5, d4
        add.w   d1, d4
        add.w   d4, d4
        adda.w  d4, a1

        ; copy d3 words
        move.w  d3, d4
        subq.w  #1, d4
.copy_row_nt:
        move.w  (a0)+, (a1)+
        dbf     d4, .copy_row_nt

        ; collision row (only if this tile row starts a new collision cell)
        move.w  d0, d4
        andi.w  #1, d4                         ; odd tile rows don't start a new collision cell
        bne.s   .done_row_coll

        lea     (Block_Stage_Collision).l, a0
        move.w  d0, d4
        lsr.w   #1, d4                         ; tile row → collision row within block
        move.w  d4, d5
        lsl.w   #4, d5                         ; collision_row * 16 (block collision stride)
        adda.w  d5, a0

        lea     (Tile_Cache_Collision).l, a1
        move.w  d2, d4
        lsr.w   #1, d4                         ; cache tile row → collision row
        move.w  d4, d5
        lsl.w   #6, d4
        lsl.w   #4, d5
        add.w   d5, d4
        add.w   d1, d4
        adda.w  d4, a1

        move.w  d3, d4
        subq.w  #1, d4
.copy_row_coll:
        move.b  (a0)+, (a1)+
        dbf     d4, .copy_row_coll
.done_row_coll:
        rts
```

**Context:** Move the VDP Sprite Attribute Table out of the scroll plane nametable area to reclaim rows 48–63 for level rendering. This gives the full 64×64 plane (512×512 pixels) for scrolling.

- [ ] **Step 1: Find a VRAM gap for SAT**

The SAT is 640 bytes (80 entries × 8 bytes). VDP register $05 encodes the SAT base address as `(address / $200)` in bits 6:0. So SAT must be $200-aligned (512-byte aligned).

Current VRAM layout:
- `$C000`: Plane A (8192 bytes, ends at $E000)
- `$D800`: SAT (current location, 640 bytes)
- `$DC00`: HScroll (896 bytes)
- `$E000`: Plane B (8192 bytes, ends at $10000)
- `$F000`: Window plane

The SAT at `$D800` sits inside Plane A's address range (`$C000-$DFFF`). Moving it out of this range frees those nametable rows.

Best candidate: `$B800` — sits in the art pool area (`$0000-$BFFF`) at the high end. This is 58 tiles worth of VRAM (tiles $5C0-$5FF). The art pool is $000-$5FF (1536 tiles), so $B800 = tile $5C0 which is within the pool. We'd lose 64 tiles from the pool but gain 16 plane rows.

Alternative: use `$D400` which is still inside the nametable range. This doesn't help.

Better: `$BC00` = tile $5E0. Loses only 32 tiles from the art pool.

Decision: `$B800` — 1472 tiles remain in pool (was 1536). The 64-tile cost is acceptable.

Update `constants.asm`:

```asm
VRAM_SPRITE_TABLE       = $B800         ; Relocated from $D800 to free plane rows 48-63
```

Actually, VDP register $05: bits 6:1 = SA15..SA10 (sprite table base address bits 15-10). Bit 0 is unused. So for address $B800: bits 15-10 = $B800 >> 10 = $2E. Register value = $2E << 1 = $5C.
